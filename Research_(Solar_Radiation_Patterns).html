# Define feature matrix for FCM
X_fcm = X_scaled.T

def xie_beni_index(X, centers, u):
    """
    Calculate Xie-Beni index for fuzzy clustering.
    Lower values indicate better clustering.
    """
    n_samples = X.shape[1]  # number of samples
    n_clusters = centers.shape[0]

    # Calculate compactness (variance within clusters)
    compactness = 0.0
    for i in range(n_clusters):
        distances = np.linalg.norm(X.T - centers[i], axis=1) ** 2
        compactness += np.sum((u[i] ** 2) * distances)

    # Calculate separation (distance between cluster centers)
    min_distance = np.inf
    for i in range(n_clusters):
        for j in range(n_clusters):
            if i != j:
                distance = np.linalg.norm(centers[i] - centers[j]) ** 2
                min_distance = min(min_distance, distance)

    # Xie-Beni index = compactness / (separation * number of samples)
    return compactness / (n_samples * min_distance)

# Test different numbers of clusters
cluster_range = range(2, 7)
pc_values = []
xb_values = []

for n_clusters in cluster_range:
    # Perform fuzzy clustering
    centers, u, _, _, _, _, fpc = fuzz.cluster.cmeans(
        X_fcm,
        c=n_clusters,
        m=2,
        error=0.005,
        maxiter=1000,
        seed=42
    )

    pc_values.append(fpc)
    xb_values.append(xie_beni_index(X_fcm, centers, u))

# Plot results
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))

# Fuzzy Partition Coefficient plot

ax1.set_xlabel("Number of Clusters")
ax1.set_ylabel("FPC Score")
ax1.grid(True, alpha=0.3)

# Xie-Beni Index plot

    print("T-tests for GHI differences between clusters")
    print("=" * 50)

    # Perform t-tests for all cluster pairs
    for cluster1, cluster2 in itertools.combinations(clusters, 2):
        ghi1 = daily[daily['fcm_cluster'] == cluster1]['ghi_mean']
        ghi2 = daily[daily['fcm_cluster'] == cluster2]['ghi_mean']

        # Perform t-test
        t_stat, p_value = stats.ttest_ind(ghi1, ghi2, equal_var=False)
